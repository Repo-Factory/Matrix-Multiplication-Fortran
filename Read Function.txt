program matrix
    implicit none
    
    real, dimension(3,3) :: array
    real :: x
    integer :: i, j
    
    open(10, file='test.txt')
    do i = 1, 3
        do j = 1, 3
            read(10, *) x
            array(i,j) = x
        end do
    end do
    
    do i = 1,3
        write(*, '(F2.0)') (array(i,j), j=1,3)
    end do

end program matrix
